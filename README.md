# Laravel deployment guide

## # add php repository 
```
sudo add-apt-repository ppa:ondrej/php

sudo apt update

sudo apt upgrade
```
## # Install php (7.1, 7.2, 7.3 and 7.4)
```
sudo apt install php7.1 php7.1-fpm php7.1-mbstring php7.1-cli php7.1-xml php7.1-bcmath php7.1-intl php7.1-sqlite3 php7.1-zip php7.1-mysql php7.1-gd php7.1-curl php7.1-pgsql php7.1-imagick
```
